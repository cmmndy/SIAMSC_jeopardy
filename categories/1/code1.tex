      \begin{algorithmic}[1]
        \FOR {$n = size(A) \cdots 2$}
        \FOR {$i = 0 \cdots n-2$}
        \IF {$A[i] > A[i+1]$}
        \STATE Swap$(A[i],A[i+1])$ 
      \ENDIF
    \ENDFOR
  \ENDFOR

\end{algorithm}
